report_builder.py
# MetaMindset Labs — PDF pipeline import pandas as pd import matplotlib.pyplot as plt from jinja2 import Environment import subprocess, os # 1. Load client data df = pd.read_excel("client_data.xlsx") # 2. Generate figures fig, ax = plt.subplots(figsize=(6, 3.5)) df.plot(ax=ax, kind="bar", color="#c8bc9e") fig.savefig("output/fig1.pdf", bbox_inches="tight") # 3. Render LaTeX template env = Environment(...) template = env.get_template("report.tex.j2") rendered = template.render(data=df.to_dict()) # 4. Write & compile with open("output/report.tex", "w") as f: f.write(rendered) subprocess.run(["pdflatex", "report.tex"]) # → report.pdf delivered to client
